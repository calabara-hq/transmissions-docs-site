import { TransmissionsClient } from '@tx-kit/sdk'
    import { CreatePublicClient, CreateWalletClient } from 'viem'
    import { base } from 'viem/chains'

    })

        chain: base,
        transport: http()
    })

    export const { uplinkClient } = new TransmissionsClient({
        publicClient,
        walletClient,
